    <script>

        let gigs = JSON.parse(localStorage.getItem('aleph_gigs')) || [];

        let settings = JSON.parse(localStorage.getItem('aleph_settings')) || { ical: '', venues: '' };


        function saveSettings() {

            settings.ical = document.getElementById('setting-ical').value;

            settings.venues = document.getElementById('setting-venues').value;

            localStorage.setItem('aleph_settings', JSON.stringify(settings));

        }


        function showView(viewId) {

            document.querySelectorAll('.tab-content').forEach(v => v.classList.remove('active'));

            document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active-nav'));

            document.getElementById('view-' + viewId).classList.add('active');

            document.getElementById('nav-' + viewId).classList.add('active-nav');

            if(viewId === 'home') renderGigs();

            if(viewId === 'settings') {

                document.getElementById('setting-ical').value = settings.ical;

                document.getElementById('setting-venues').value = settings.venues;

            }

        }


        // --- CALENDAR SYNC LOGIC ---

        async function syncCalendar() {

            if(!settings.ical) return alert("Add your iCal link in Settings first!");

            

            try {

                // Using a CORS proxy to allow browser to read Google's feed

                const proxy = "https://corsproxy.io/?";

                const response = await fetch(proxy + encodeURIComponent(settings.ical));

                const text = await response.text();

                

                // Simple ICS parser

                const events = text.split("BEGIN:VEVENT");

                events.shift(); // remove header


                events.forEach(ev => {

                    const title = ev.match(/SUMMARY:(.*)/)?.[1]?.trim();

                    const rawStart = ev.match(/DTSTART[:;](?:VALUE=DATE:)?(\d+T\d+|\d+)/)?.[1];

                    const location = ev.match(/LOCATION:(.*)/)?.[1]?.trim() || "Venue TBC";

                    

                    if(title && rawStart) {

                        const dateStr = `${rawStart.substring(0,4)}-${rawStart.substring(4,6)}-${rawStart.substring(6,8)}`;

                        const timeStr = rawStart.includes('T') ? `${rawStart.substring(9,11)}:${rawStart.substring(11,13)}` : "20:00";

                        

                        // Check if gig already exists

                        if(!gigs.find(g => g.title === title && g.date === dateStr)) {

                            const venueData = matchVenue(location || title);

                            gigs.push({

                                id: Date.now() + Math.random(),

                                title, venue: location, date: dateStr, time: timeStr,

                                cap: venueData.cap, link: venueData.link, sold: 0

                            });

                        }

                    }

                });

                localStorage.setItem('aleph_gigs', JSON.stringify(gigs));

                renderGigs();

                alert("Gigs Updated from Google!");

            } catch(e) { alert("Sync failed. Check your link."); }

        }


        function matchVenue(name) {

            const lines = settings.venues.split('\n');

            for(let line of lines) {

                const [vName, cap, link] = line.split(',').map(s => s.trim());

                if(name.toLowerCase().includes(vName.toLowerCase())) {

                    return { cap: parseInt(cap) || 50, link: link || '' };

                }

            }

            return { cap: 50, link: '' };

        }


        function renderGigs() {

            const list = document.getElementById('gig-list');

            list.innerHTML = '';

            gigs.sort((a,b) => new Date(a.date) - new Date(b.date)).forEach(gig => {

                const isToday = new Date().toISOString().split('T')[0] === gig.date;

                list.innerHTML += `

                    <div class="card ${isToday ? 'border-blue-500 bg-blue-900/20' : ''}" onclick="viewDetail(${gig.id})">

                        <div class="flex justify-between items-center">

                            <div>

                                <h2 class="font-bold text-lg">${gig.title}</h2>

                                <p class="text-slate-400 text-xs">${gig.venue} • ${gig.date}</p>

                            </div>

                            <div class="text-right">

                                <span class="text-xs font-mono text-blue-400">${gig.time}</span>

                            </div>

                        </div>

                    </div>`;

            });

        }


        function viewDetail(id) {

            const gig = gigs.find(g => g.id === id);

            const ratio = gig.sold / gig.cap;

            const scarcity = ratio > 0.8 ? "🚨 SOLD OUT/LOW TICKETS" : "Tickets Available";

            

            document.getElementById('detail-content').innerHTML = `

                <div class="card border-l-4 border-blue-500">

                    <h2 class="text-2xl font-black">${gig.title}</h2>

                    <p class="text-blue-400 font-bold mb-4">${gig.venue}</p>

                    <div class="flex justify-between text-xs mb-1">

                        <span>Sold: ${gig.sold}</span>

                        <span>Cap: ${gig.cap}</span>

                    </div>

                    <input type="range" min="0" max="${gig.cap}" value="${gig.sold}" onchange="updateSold(${id}, this.value)" class="w-full h-2 bg-slate-700 rounded-lg appearance-none cursor-pointer">

                </div>


                <div class="card bg-slate-800">

                    <h3 class="text-xs font-bold text-slate-500 uppercase tracking-widest mb-2">Promo Caption</h3>

                    <div class="text-sm leading-relaxed" id="cap-box">

                        Next up: ${gig.title}<br>

                        📍 ${gig.venue}<br>

                        🗓️ ${gig.date} | ${gig.time}<br><br>

                        ${ratio > 0.8 ? '🔥 Low Ticket Alert!' : 'I would love to see you there.'}<br>

                        Tickets: ${gig.link || '[Link]'}

                    </div>

                    <button onclick="copyCap()" class="btn-primary mt-4 bg-slate-700 text-sm py-2">Copy for Instagram</button>

                </div>

                <button onclick="deleteGig(${id})" class="w-full text-slate-600 text-xs mt-4">Remove Gig</button>

            `;

            showView('detail');

        }


        function updateSold(id, val) {

            const gig = gigs.find(g => g.id === id);

            gig.sold = parseInt(val);

            localStorage.setItem('aleph_gigs', JSON.stringify(gigs));

            viewDetail(id);

        }


        function deleteGig(id) { if(confirm("Delete?")) { gigs = gigs.filter(g => g.id !== id); localStorage.setItem('aleph_gigs', JSON.stringify(gigs)); showView('home'); } }

        

        function copyCap() {

            const text = document.getElementById('cap-box').innerText;

            navigator.clipboard.writeText(text);

            alert("Copied to clipboard!");

        }


        renderGigs();

    </script>
